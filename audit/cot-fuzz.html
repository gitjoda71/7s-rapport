const out = document.getElementById('cases');
let allOk = true;
CASES.forEach(c => {
  const xml = makeCoT(c.input);
  const r = check(xml);
  if (!r.ok) allOk = false;
  // Bonus-check: -text-noden ska innehålla exakt input efter unescape
  let textOk = false;
  if (r.dom) {
    const remarksText = r.dom.querySelector('remarks')?.textContent || '';
    // null/undefined coerce:as till '' av e(); sedan `${remarks || '-'}` → '-'.
    // Andra non-strings (number, boolean, object, array) String:ifieras.
    const coerced = (c.input == null) ? '' : String(c.input);
    const expected = '7S RAPPORT — ' + (coerced || '-');
    textOk = remarksText === expected;
  }
  const div = document.createElement('div');
  div.className = 'case';
  div.innerHTML =
    '
Test' + c.name + '
' +
    '
Input' + e(c.input) + '
' +
    '
Resultat' +
      (r.ok ? '✓ XML giltig, ' : '✗ XML ogiltig: ' + r.why + ', ') +
      (textOk ? '✓ remarks-text bevarad' : '✗ remarks-text avviker') +
    '
' +
    '
Visa producerad XML
' + e(xml) + '
';
  out.appendChild(div);
});

const banner = document.createElement('div');
banner.className = 'case';
banner.innerHTML = '' +
  (allOk ? '✓ Alla auto-tester gick igenom — escape:n håller.' : '✗ Minst ett auto-test misslyckades. Se ovan.') + '';
out.prepend(banner);
