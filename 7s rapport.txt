<body>
    <div class="container">
        <div class="header">
            <h1>7S Rapport</h1>
        </div>

        <!-- Formulärvy -->
        <div class="form-view" id="formView">
            
            <!-- STUND -->
            <div class="form-group">
                <div class="label-row">
                    <label>Stund</label>
                    <button class="info-btn" tabindex="-1" onclick="toggleInfo('infoStund')">i</button>
                </div>
                <div class="info-text" id="infoStund">Tidpunkt när det hände. Format: DDHHMM (dag, timme, minut)</div>
                <div class="input-row">
                    <input type="text" id="stund" class="stund-input" maxlength="6" placeholder="DDHHMM">
                    <button class="btn btn-time" tabindex="-1" onclick="setCurrentTime()">Nu</button>
                </div>
            </div>

            <!-- STÄLLE -->
            <div class="form-group">
                <div class="label-row">
                    <label>Ställe</label>
                    <button class="info-btn" tabindex="-1" onclick="toggleInfo('infoStalle')">i</button>
                </div>
                <div class="info-text" id="infoStalle">Plats där det hände. Koordinater, platsnamn eller beskrivning.</div>
                <input type="text" id="stalle" placeholder="Plats / koordinater">
                <div class="saved-items" id="savedStalle"></div>
            </div>

            <!-- STYRKA -->
            <div class="form-group">
                <div class="label-row">
                    <label>Styrka</label>
                    <button class="info-btn" tabindex="-1" onclick="toggleInfo('infoStyrka')">i</button>
                </div>
                <div class="info-text" id="infoStyrka">Antal observerade fiender, fordon eller objekt.</div>
                <input type="text" id="styrka" placeholder="Antal">
            </div>

            <!-- SLAG -->
            <div class="form-group">
                <div class="label-row">
                    <label>Slag</label>
                    <button class="info-btn" tabindex="-1" onclick="toggleInfo('infoSlag')">i</button>
                </div>
                <div class="info-text" id="infoSlag">Typ av fiender eller objekt. T.ex. stridsvagn, infanteri, lastbil.</div>
                <textarea id="slag" rows="2" placeholder="Typ av fiende/objekt"></textarea>
            </div>

            <!-- SYSSELSÄTTNING -->
            <div class="form-group">
                <div class="label-row">
                    <label>Sysselsättning</label>
                    <button class="info-btn" tabindex="-1" onclick="toggleInfo('infoSyssel')">i</button>
                </div>
                <div class="info-text" id="infoSyssel">Fiendens verksamhet. Vad gör de? Rörelser, gruppering, etc.</div>
                <textarea id="sysselsattning" rows="2" placeholder="Vad gör fienden?"></textarea>
            </div>

            <!-- SYMBOL -->
            <div class="form-group">
                <div class="label-row">
                    <label>Symbol</label>
                    <button class="info-btn" tabindex="-1" onclick="toggleInfo('infoSymbol')">i</button>
                </div>
                <div class="info-text" id="infoSymbol">Märkning, siffror, förbandstecken, färg på fordon/uniformer.</div>
                <textarea id="symbol" rows="2" placeholder="Märkning, färg, tecken"></textarea>
            </div>

            <!-- SAGESMAN -->
            <div class="form-group">
                <div class="label-row">
                    <label>Sagesman</label>
                    <button class="info-btn" tabindex="-1" onclick="toggleInfo('infoSagesman')">i</button>
                </div>
                <div class="info-text" id="infoSagesman">Vem som har gjort observationen.</div>
                <input type="text" id="sagesman" placeholder="Observatör">
                <div class="saved-items" id="savedSagesman"></div>
            </div>

            <hr class="divider">

            <button class="btn btn-primary" onclick="generateReport()">Generera rapport</button>
        </div>

        <!-- Resultatvy -->
        <div class="result-view" id="resultView">
            <div class="result-header">Rapport redo att kopieras</div>
            
            <div class="report-output" id="reportOutput"></div>
            
            <button class="btn btn-copy" onclick="copyToClipboard()">Kopiera till urklipp</button>
            <div class="copy-feedback" id="copyFeedback">✓ Kopierat</div>
            
            <div class="button-group">
                <button class="btn btn-secondary" onclick="backToEdit()">Redigera</button>
                <button class="btn btn-danger" onclick="resetForm()">Nollställ</button>
            </div>
        </div>
    </div>

    <script>
        document.addEventListener('DOMContentLoaded', function() {
            loadSavedItems();
            loadLastSagesman();
        });

        function toggleInfo(id) {
            const el = document.getElementById(id);
            el.classList.toggle('show');
        }

        function setCurrentTime() {
            const now = new Date();
            const day = String(now.getDate()).padStart(2, '0');
            const hours = String(now.getHours()).padStart(2, '0');
            const minutes = String(now.getMinutes()).padStart(2, '0');
            document.getElementById('stund').value = day + hours + minutes;
        }

        function generateReport() {
            const stund = document.getElementById('stund').value;
            const stalle = document.getElementById('stalle').value;
            const styrka = document.getElementById('styrka').value;
            const slag = document.getElementById('slag').value;
            const sysselsattning = document.getElementById('sysselsattning').value;
            const symbol = document.getElementById('symbol').value;
            const sagesman = document.getElementById('sagesman').value;

            savePlaceIfNew(stalle);
            saveSagesmanIfNew(sagesman);

            let report = "7S RAPPORT\n\n";
            report += `Stund: ${stund || '-'}\n`;
            report += `Ställe: ${stalle || '-'}\n`;
            report += `Styrka: ${styrka || '-'}\n`;
            report += `Slag: ${slag || '-'}\n`;
            report += `Sysselsättning: ${sysselsattning || '-'}\n`;
            report += `Symbol: ${symbol || '-'}\n`;
            report += `Sagesman: ${sagesman || '-'}`;

            document.getElementById('reportOutput').textContent = report;
            document.getElementById('formView').classList.add('hidden');
            document.getElementById('resultView').classList.add('active');
        }

        function copyToClipboard() {
            const reportText = document.getElementById('reportOutput').textContent;
            
            if (navigator.clipboard && navigator.clipboard.writeText) {
                navigator.clipboard.writeText(reportText).then(function() {
                    showCopyFeedback();
                }).catch(function() {
                    fallbackCopy(reportText);
                });
            } else {
                fallbackCopy(reportText);
            }
        }

        function fallbackCopy(text) {
            const textarea = document.createElement('textarea');
            textarea.value = text;
            textarea.style.position = 'fixed';
            textarea.style.opacity = '0';
            document.body.appendChild(textarea);
            textarea.select();
            try {
                document.execCommand('copy');
                showCopyFeedback();
            } catch (e) {
                alert('Kunde inte kopiera. Markera texten manuellt.');
            }
            document.body.removeChild(textarea);
        }

        function showCopyFeedback() {
            const feedback = document.getElementById('copyFeedback');
            feedback.classList.remove('show');
            void feedback.offsetWidth;
            feedback.classList.add('show');
        }

        function backToEdit() {
            document.getElementById('formView').classList.remove('hidden');
            document.getElementById('resultView').classList.remove('active');
        }

        function resetForm() {
            if (confirm('Nollställ rapporten?')) {
                document.getElementById('stund').value = '';
                document.getElementById('stalle').value = '';
                document.getElementById('styrka').value = '';
                document.getElementById('slag').value = '';
                document.getElementById('sysselsattning').value = '';
                document.getElementById('symbol').value = '';
                backToEdit();
            }
        }

        function getSavedPlaces() {
            const saved = localStorage.getItem('7s_places');
            return saved ? JSON.parse(saved) : [];
        }

        function getSavedSagesman() {
            const saved = localStorage.getItem('7s_sagesman');
            return saved ? JSON.parse(saved) : [];
        }

        function savePlaceIfNew(place) {
            if (!place || place.trim() === '') return;
            let places = getSavedPlaces();
            if (!places.includes(place)) {
                places.unshift(place);
                places = places.slice(0, 5);
                localStorage.setItem('7s_places', JSON.stringify(places));
                loadSavedItems();
            }
        }

        function saveSagesmanIfNew(name) {
            if (!name || name.trim() === '') return;
            let names = getSavedSagesman();
            if (!names.includes(name)) {
                names.unshift(name);
                names = names.slice(0, 5);
                localStorage.setItem('7s_sagesman', JSON.stringify(names));
            }
            localStorage.setItem('7s_lastSagesman', name);
            loadSavedItems();
        }

        function loadLastSagesman() {
            const last = localStorage.getItem('7s_lastSagesman');
            if (last) {
                document.getElementById('sagesman').value = last;
            }
        }

        function loadSavedItems() {
            const places = getSavedPlaces();
            const placeContainer = document.getElementById('savedStalle');
            placeContainer.innerHTML = '';
            places.forEach(place => {
                const span = document.createElement('span');
                span.className = 'saved-item';
                span.innerHTML = place + '<span class="delete" onclick="event.stopPropagation(); deletePlace(\'' + place.replace(/'/g, "\\'") + '\')">×</span>';
                span.onclick = () => document.getElementById('stalle').value = place;
                placeContainer.appendChild(span);
            });

            const names = getSavedSagesman();
            const nameContainer = document.getElementById('savedSagesman');
            nameContainer.innerHTML = '';
            names.forEach(name => {
                const span = document.createElement('span');
                span.className = 'saved-item';
                span.innerHTML = name + '<span class="delete" onclick="event.stopPropagation(); deleteSagesman(\'' + name.replace(/'/g, "\\'") + '\')">×</span>';
                span.onclick = () => document.getElementById('sagesman').value = name;
                nameContainer.appendChild(span);
            });
        }

        function deletePlace(place) {
            let places = getSavedPlaces();
            places = places.filter(p => p !== place);
            localStorage.setItem('7s_places', JSON.stringify(places));
            loadSavedItems();
        }

        function deleteSagesman(name) {
            let names = getSavedSagesman();
            names = names.filter(n => n !== name);
            localStorage.setItem('7s_sagesman', JSON.stringify(names));
            loadSavedItems();
        }
    </script>

</body>
